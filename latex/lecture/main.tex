\end{definition}

\begin{theorem}{Fundamental Theorem of Calculus}{ftc}
    If $f$ is continuous on $[a, b]$, then:
    \[
        \int_a^b f(x) \, dx = F(b) - F(a)
    \]
\end{theorem}
